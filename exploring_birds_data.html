##is there a similar monthly trend every year in bird collision count ?

bird_collision %>% 
  separate(col="date", into=c("month", "day", "year"), sep = "/") %>% 
  select(year, month, count) %>% 
  mutate(month=as.factor(month)) %>% 
  group_by(year, month) %>% 
  summarize(monthly_collision = sum(count), .groups = "drop") %>% 
  ggplot(aes(x = month, y = monthly_collision)) +
  geom_col(aes(fill = month)) +
  facet_grid(~year) + 
  labs ( title = "Monthly Bird Collisions by year", x = "Month", y = "Total Collisions")

##does having a east wind (east to west) result in different # of collisions than west wind (west to east)

bird_collision %>% 
  select(uwind, count) %>% 
  mutate(wind_dir = ifelse(uwind > 0, "east", "west"),
         wind_dir = as.factor(wind_dir)) %>% 
  group_by(wind_dir) %>% 
  summarize(total_collision = sum(count), .groups = "drop") %>% 
  ggplot(aes(x = wind_dir, y = total_collision)) + 
  geom_col(aes(fill = wind_dir)) + 
  labs (title = "Wind Direction and Collision count", x = "Wind Direction", y = "Total Collision", fill = "Wind Direction")
